=== OVERVIEW ===
The @reactjit/apis package includes hooks for both Plex (token-based) and Jellyfin/Emby (API key-based) self-hosted media servers. All hooks are passive — they fetch on mount and can be polled automatically. Pass null for any credential argument to suppress fetching entirely, which makes it safe to render components before credentials are available.

Plex hooks use the X-Plex-Token header. Jellyfin and Emby hooks use the MediaBrowser Token authorization header. Both server families are covered by a single import.

=== API / SYNTAX ===

--- usePlexLibraries ---

Fetch all library sections configured on the Plex server.

  import { usePlexLibraries } from '@reactjit/apis';

  const result: APIResult<{ MediaContainer: { Directory: PlexLibrary[] } }> = usePlexLibraries(
    baseUrl: string | null,
    token: string | null,
  );

Parameters:
  baseUrl  string | null  Base URL of the Plex server (e.g. 'http://localhost:32400')
  token    string | null  Plex authentication token

Returns: APIResult<{ MediaContainer: { Directory: PlexLibrary[] } }>

PlexLibrary fields:
  key      string  Section ID (used in usePlexLibrary)
  title    string  Human-readable library name
  type     string  'movie', 'show', 'music', or 'photo'
  agent    string  Metadata agent identifier
  scanner  string  Scanner identifier

--- usePlexLibrary ---

Fetch all items in a specific library section.

  import { usePlexLibrary } from '@reactjit/apis';

  const result: APIResult<{ MediaContainer: { Metadata: PlexMediaItem[] } }> = usePlexLibrary(
    baseUrl: string | null,
    token: string | null,
    sectionId: string | null,
  );

Parameters:
  baseUrl    string | null  Base URL of the Plex server
  token      string | null  Plex authentication token
  sectionId  string | null  Library section key from usePlexLibraries

Returns: APIResult<{ MediaContainer: { Metadata: PlexMediaItem[] } }>

PlexMediaItem fields:
  ratingKey      string   Unique item identifier
  title          string   Item title
  type           string   'movie', 'show', 'episode', etc.
  year           number | undefined
  thumb          string | undefined  Thumbnail path (append to baseUrl)
  art            string | undefined  Background art path
  summary        string | undefined
  duration       number | undefined  Duration in milliseconds
  addedAt        number   Unix timestamp
  updatedAt      number   Unix timestamp
  viewCount      number | undefined  Times played
  lastViewedAt   number | undefined  Unix timestamp of last play

--- usePlexRecentlyAdded ---

Fetch the most recently added items across all libraries.

  import { usePlexRecentlyAdded } from '@reactjit/apis';

  const result: APIResult<{ MediaContainer: { Metadata: PlexMediaItem[] } }> = usePlexRecentlyAdded(
    baseUrl: string | null,
    token: string | null,
    opts?: { limit?: number },
  );

Parameters:
  baseUrl  string | null   Base URL of the Plex server
  token    string | null   Plex authentication token
  opts     object | undefined
    limit  number          Number of items to return (default: 20)

Returns: APIResult<{ MediaContainer: { Metadata: PlexMediaItem[] } }>

--- usePlexSessions ---

Fetch currently active playback sessions. Supports automatic polling.

  import { usePlexSessions } from '@reactjit/apis';

  const result: APIResult<{ MediaContainer: { Metadata?: PlexSession[] } }> = usePlexSessions(
    baseUrl: string | null,
    token: string | null,
    opts?: { interval?: number },
  );

Parameters:
  baseUrl  string | null   Base URL of the Plex server
  token    string | null   Plex authentication token
  opts     object | undefined
    interval  number       Polling interval in milliseconds (default: 10000)

Returns: APIResult<{ MediaContainer: { Metadata?: PlexSession[] } }>
  Metadata is undefined when no sessions are active.

PlexSession fields:
  sessionKey   string   Unique session identifier
  title        string   Title of the playing item
  type         string   Media type
  thumb        string | undefined  Thumbnail path
  User         { title: string }
  Player       { title: string; state: string; platform: string }
               state is 'playing', 'paused', or 'buffering'
  viewOffset   number   Current position in milliseconds
  duration     number   Total duration in milliseconds

--- useJellyfinLibraries ---

Fetch all virtual library folders from a Jellyfin or Emby server.

  import { useJellyfinLibraries } from '@reactjit/apis';

  const result: APIResult<{ Items: JellyfinLibrary[] }> = useJellyfinLibraries(
    baseUrl: string | null,
    apiKey: string | null,
  );

Parameters:
  baseUrl  string | null  Base URL of the Jellyfin/Emby server
  apiKey   string | null  API key from the server dashboard

Returns: APIResult<{ Items: JellyfinLibrary[] }>

JellyfinLibrary fields:
  Id              string
  Name            string
  CollectionType  string   'movies', 'tvshows', 'music', etc.
  ImageTags       Record<string, string> | undefined

--- useJellyfinItems ---

Query media items with flexible filtering and sorting.

  import { useJellyfinItems } from '@reactjit/apis';

  const result: APIResult<{ Items: JellyfinItem[]; TotalRecordCount: number }> = useJellyfinItems(
    baseUrl: string | null,
    apiKey: string | null,
    opts?: { parentId?: string; type?: string; limit?: number; sortBy?: string },
  );

Parameters:
  baseUrl  string | null   Base URL of the Jellyfin/Emby server
  apiKey   string | null   API key
  opts     object | undefined
    parentId  string       Filter to a specific library or folder
    type      string       Item type filter (e.g. 'Movie', 'Series')
    limit     number       Max items to return (default: 20)
    sortBy    string       Sort field (default: 'DateCreated')

Returns: APIResult<{ Items: JellyfinItem[]; TotalRecordCount: number }>

JellyfinItem fields:
  Id               string
  Name             string
  Type             string
  Overview         string | undefined
  ProductionYear   number | undefined
  ImageTags        Record<string, string> | undefined
  RunTimeTicks     number | undefined  Duration in 100-nanosecond ticks
  CommunityRating  number | undefined
  DateCreated      string | undefined

--- useJellyfinSessions ---

Fetch currently active playback sessions from a Jellyfin/Emby server. Supports polling.

  import { useJellyfinSessions } from '@reactjit/apis';

  const result: APIResult<JellyfinSession[]> = useJellyfinSessions(
    baseUrl: string | null,
    apiKey: string | null,
    opts?: { interval?: number },
  );

Parameters:
  baseUrl  string | null   Base URL of the server
  apiKey   string | null   API key
  opts     object | undefined
    interval  number       Polling interval in milliseconds (default: 10000)

Returns: APIResult<JellyfinSession[]>

JellyfinSession fields:
  Id               string
  UserName         string
  Client           string
  DeviceName       string
  NowPlayingItem   JellyfinItem | undefined  Absent when idle
  PlayState        { PositionTicks: number; IsPaused: boolean } | undefined

--- useJellyfinLatest ---

Fetch the latest items a user has added or accessed. Requires a user ID.

  import { useJellyfinLatest } from '@reactjit/apis';

  const result: APIResult<JellyfinItem[]> = useJellyfinLatest(
    baseUrl: string | null,
    apiKey: string | null,
    userId: string | null,
    opts?: { limit?: number },
  );

Parameters:
  baseUrl  string | null   Base URL of the server
  apiKey   string | null   API key
  userId   string | null   Jellyfin user ID
  opts     object | undefined
    limit  number          Max items to return (default: 20)

Returns: APIResult<JellyfinItem[]>

--- jellyfinImage (utility) ---

Build a Jellyfin image URL. Not a hook — call it directly in JSX.

  import { jellyfinImage } from '@reactjit/apis';

  const url: string = jellyfinImage(
    baseUrl: string,
    itemId: string,
    tag: string,
    type?: string,
    maxWidth?: number,
  );

Parameters:
  baseUrl   string  Base URL of the server
  itemId    string  Item ID
  tag       string  Image tag from ImageTags
  type      string  Image type (default: 'Primary')
  maxWidth  number  Max image width for server-side resize (default: 400)

Returns: string — ready-to-use image URL for use in <Image src={...} />

=== EXAMPLES ===

-- Plex now-playing widget --

Poll active sessions every 10 seconds and display who is watching what:

  import { usePlexSessions } from '@reactjit/apis';
  import { Box, Text } from '@reactjit/core';

  const PLEX_URL = 'http://192.168.1.10:32400';
  const PLEX_TOKEN = process.env.PLEX_TOKEN;

  function NowPlaying() {
    const { data, loading } = usePlexSessions(PLEX_URL, PLEX_TOKEN, { interval: 10000 });
    const sessions = data?.MediaContainer?.Metadata ?? [];

    if (loading && sessions.length === 0) {
      return <Text fontSize={13}>Connecting to Plex...</Text>;
    }

    if (sessions.length === 0) {
      return <Text fontSize={13}>Nothing playing</Text>;
    }

    return (
      <Box gap={8}>
        {sessions.map((s) => {
          const pct = Math.round((s.viewOffset / s.duration) * 100);
          return (
            <Box key={s.sessionKey} padding={12} backgroundColor="#1a1a2e">
              <Text fontSize={14}>{s.title}</Text>
              <Text fontSize={12}>{s.User.title} on {s.Player.title}</Text>
              <Text fontSize={11}>{pct}% — {s.Player.state}</Text>
            </Box>
          );
        })}
      </Box>
    );
  }

-- Jellyfin recently added shelf --

Show the 10 most recently added movies with poster images:

  import { useJellyfinItems, jellyfinImage } from '@reactjit/apis';
  import { Box, Text, Image } from '@reactjit/core';

  const JELLY_URL = 'http://192.168.1.20:8096';
  const JELLY_KEY = process.env.JELLYFIN_API_KEY;

  function RecentMovies() {
    const { data, loading } = useJellyfinItems(JELLY_URL, JELLY_KEY, {
      type: 'Movie',
      limit: 10,
      sortBy: 'DateCreated',
    });

    if (loading) return <Text fontSize={13}>Loading...</Text>;

    return (
      <Box flexDirection="row" gap={12} flexWrap="wrap">
        {(data?.Items ?? []).map((item) => {
          const tag = item.ImageTags?.Primary;
          return (
            <Box key={item.Id} width={120}>
              {tag && (
                <Image
                  src={jellyfinImage(JELLY_URL, item.Id, tag, 'Primary', 240)}
                  width={120}
                  height={180}
                />
              )}
              <Text fontSize={11}>{item.Name}</Text>
              <Text fontSize={10}>{item.ProductionYear}</Text>
            </Box>
          );
        })}
      </Box>
    );
  }
