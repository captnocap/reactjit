=== OVERVIEW ===
The CoinGecko hooks cover the full public CoinGecko v3 API: spot prices, market lists, coin detail pages, historical OHLCV data, trending coin lists, and a global market summary. All hooks are read-only and built on the reactive `useAPI` base — they fetch on mount and can be polled on an interval.

Auth: The free public tier requires no API key. Pass your CoinGecko Demo API key as `opts.apiKey` to increase rate limits. The key is sent as the `x-cg-demo-api-key` request header.

=== API / SYNTAX ===

--- useCoinPrice ---

Fetch the spot price of one or more coins in any supported currency. Returns a flat dictionary keyed by coin ID.

  import { useCoinPrice } from '@reactjit/apis/coingecko';

  const result: APIResult<CoinPrice> = useCoinPrice(
    coinIds: string | string[] | null,
    opts?:   {
      vs?:              string;
      include24hChange?: boolean;
      apiKey?:          string | null;
    },
  );

Parameters:
  coinIds              string | string[] | null   One or more CoinGecko coin IDs (e.g. 'bitcoin', ['bitcoin','ethereum']). Pass null to disable.
  opts.vs              string                     Target currency. Defaults to 'usd'.
  opts.include24hChange boolean                   Include 24-hour percent change. Defaults to true.
  opts.apiKey          string | null              Optional Demo API key for higher rate limits.

Returns: APIResult<CoinPrice>
  data  CoinPrice | null

CoinPrice shape:
  {
    bitcoin: {
      usd: 65000,
      usd_24h_change: -1.24,
    },
    ethereum: {
      usd: 3100,
      usd_24h_change: 0.87,
    },
  }

--- useCoinMarkets ---

Fetch a paginated, market-cap-sorted list of coins with price statistics and optional 7-day sparkline data.

  import { useCoinMarkets } from '@reactjit/apis/coingecko';

  const result: APIResult<CoinMarket[]> = useCoinMarkets(
    opts?: {
      vs?:       string;
      perPage?:  number;
      page?:     number;
      sparkline?: boolean;
      apiKey?:   string | null;
    },
  );

Parameters:
  opts.vs        string    Target currency. Defaults to 'usd'.
  opts.perPage   number    Results per page. Defaults to 20. Max 250.
  opts.page      number    Page number. Defaults to 1.
  opts.sparkline boolean   Include 7-day sparkline price array. Defaults to false.
  opts.apiKey    string | null

Returns: APIResult<CoinMarket[]>
  data  CoinMarket[] | null

CoinMarket fields:
  id                             string
  symbol                         string
  name                           string
  image                          string             Logo URL
  current_price                  number
  market_cap                     number
  market_cap_rank                number
  total_volume                   number
  high_24h                       number
  low_24h                        number
  price_change_24h               number
  price_change_percentage_24h    number
  circulating_supply             number
  total_supply                   number | null
  ath                            number             All-time high price
  ath_date                       string
  atl                            number             All-time low price
  atl_date                       string
  sparkline_in_7d                { price: number[] } | undefined   Only when opts.sparkline is true

--- useCoinDetail ---

Fetch detailed information about a single coin: description, market data across all timeframes, links, and categories.

  import { useCoinDetail } from '@reactjit/apis/coingecko';

  const result: APIResult<CoinDetail> = useCoinDetail(
    coinId: string | null,
    opts?:  { apiKey?: string | null },
  );

Parameters:
  coinId    string | null   CoinGecko coin ID (e.g. 'bitcoin'). Pass null to disable.
  opts.apiKey  string | null

Returns: APIResult<CoinDetail>
  data  CoinDetail | null

CoinDetail fields:
  id                                    string
  symbol                                string
  name                                  string
  description.en                        string              English text description
  image.thumb / small / large           string              Logo image URLs
  market_data.current_price             Record<string, number>
  market_data.market_cap                Record<string, number>
  market_data.total_volume              Record<string, number>
  market_data.high_24h                  Record<string, number>
  market_data.low_24h                   Record<string, number>
  market_data.price_change_percentage_24h   number
  market_data.price_change_percentage_7d    number
  market_data.price_change_percentage_30d   number
  links.homepage                        string[]
  links.blockchain_site                 string[]
  categories                            string[]

--- useCoinHistory ---

Fetch OHLCV-style historical price, market cap, and volume data for a coin over a time window. Returns timestamp-value pairs.

  import { useCoinHistory } from '@reactjit/apis/coingecko';

  const result: APIResult<CoinHistory> = useCoinHistory(
    coinId: string | null,
    opts?:  { vs?: string; days?: number | 'max'; apiKey?: string | null },
  );

Parameters:
  coinId    string | null          CoinGecko coin ID. Pass null to disable.
  opts.vs   string                 Target currency. Defaults to 'usd'.
  opts.days number | 'max'         Lookback window in days, or 'max' for all history. Defaults to 7.
  opts.apiKey  string | null

Returns: APIResult<CoinHistory>
  data.prices        Array<[timestamp: number, price: number]>
  data.market_caps   Array<[timestamp: number, cap: number]>
  data.total_volumes Array<[timestamp: number, volume: number]>

--- useCoinTrending ---

Fetch the top 7 trending coins on CoinGecko in the last 24 hours (based on search volume).

  import { useCoinTrending } from '@reactjit/apis/coingecko';

  const result: APIResult<CoinTrending> = useCoinTrending(
    opts?: { apiKey?: string | null },
  );

Parameters:
  opts.apiKey  string | null

Returns: APIResult<CoinTrending>
  data.coins  Array of trending item wrappers

Each item in data.coins:
  item.id        string
  item.coin_id   number
  item.name      string
  item.symbol    string
  item.thumb     string   Small logo URL
  item.score     number   Trending rank (0 = top)
  item.price_btc number   Price in BTC

--- useCoinGlobal ---

Fetch aggregate global cryptocurrency market statistics: total market cap, volume, dominance percentages, and 24-hour change.

  import { useCoinGlobal } from '@reactjit/apis/coingecko';

  const result: APIResult<CoinGlobal> = useCoinGlobal(
    opts?: { apiKey?: string | null },
  );

Parameters:
  opts.apiKey  string | null

Returns: APIResult<CoinGlobal>
  data.data.active_cryptocurrencies                  number
  data.data.markets                                  number   Number of exchanges tracked
  data.data.total_market_cap                         Record<string, number>
  data.data.total_volume                             Record<string, number>
  data.data.market_cap_percentage                    Record<string, number>   Dominance per coin
  data.data.market_cap_change_percentage_24h_usd     number

=== EXAMPLES ===

-- Live price ticker with 24-hour change --

Poll BTC and ETH prices every 30 seconds and display them with color-coded change indicators.

  import { useCoinPrice } from '@reactjit/apis/coingecko';
  import { Box, Text } from '@reactjit/core';

  const COINS = ['bitcoin', 'ethereum', 'solana'];

  function PriceTicker() {
    // Poll every 30 seconds — respect free tier limits
    const { data, loading, error } = useCoinPrice(COINS, { vs: 'usd' });

    if (loading && !data) return <Text fontSize={14}>Loading prices...</Text>;
    if (error) return <Text fontSize={14}>Error: {error.message}</Text>;

    return (
      <Box width="100%" gap={4} padding={12}>
        {COINS.map((id) => {
          const entry = data?.[id];
          const price = entry?.usd ?? 0;
          const change = entry?.usd_24h_change ?? 0;
          const up = change >= 0;

          return (
            <Box key={id} flexDirection="row" gap={12} alignItems="center">
              <Text fontSize={14}>{id.toUpperCase()}</Text>
              <Text fontSize={14}>${price.toLocaleString()}</Text>
              <Text fontSize={12} color={up ? '#4CAF50' : '#f44336'}>
                {up ? '+' : ''}{change.toFixed(2)}%
              </Text>
            </Box>
          );
        })}
      </Box>
    );
  }

-- Top 10 coins by market cap with sparklines --

Fetch the top 10 coins with sparkline data and render a miniature 7-day price line using the Sparkline component.

  import { useCoinMarkets } from '@reactjit/apis/coingecko';
  import { Sparkline } from '@reactjit/shared';
  import { Box, Text } from '@reactjit/core';

  function MarketTable() {
    const { data, loading } = useCoinMarkets({ perPage: 10, sparkline: true });

    if (loading || !data) return <Text fontSize={14}>Loading market data...</Text>;

    return (
      <Box width="100%" gap={6} padding={12}>
        <Box flexDirection="row" gap={8}>
          <Text fontSize={12}>#</Text>
          <Text fontSize={12}>Coin</Text>
          <Text fontSize={12}>Price</Text>
          <Text fontSize={12}>24h</Text>
          <Text fontSize={12}>7d</Text>
        </Box>
        {data.map((coin) => {
          const up = coin.price_change_percentage_24h >= 0;
          return (
            <Box key={coin.id} flexDirection="row" gap={8} alignItems="center" padding={6}>
              <Text fontSize={12}>{coin.market_cap_rank}</Text>
              <Text fontSize={13}>{coin.symbol.toUpperCase()}</Text>
              <Text fontSize={12}>${coin.current_price.toLocaleString()}</Text>
              <Text fontSize={11} color={up ? '#4CAF50' : '#f44336'}>
                {up ? '+' : ''}{coin.price_change_percentage_24h.toFixed(1)}%
              </Text>
              {coin.sparkline_in_7d && (
                <Sparkline
                  data={coin.sparkline_in_7d.price}
                  width={80}
                  height={28}
                  color={up ? '#4CAF50' : '#f44336'}
                />
              )}
            </Box>
          );
        })}
      </Box>
    );
  }
