=== OVERVIEW ===
The Steam hooks wrap the official Steam Web API and the Steam Store API. Use them to build game library browsers, playtime dashboards, achievement trackers, or friend activity panels. All hooks are read-only and built on the reactive `useAPI` base.

Auth: Get a Web API key at https://steamcommunity.com/dev/apikey. The key is sent as a query parameter (`key=...`). Note that some endpoints (friend list, recently played games) require the target profile to be set to public visibility.

=== API / SYNTAX ===

--- useSteamUser ---

Fetch public profile information for a Steam user by their 64-bit SteamID.

  import { useSteamUser } from '@reactjit/apis/steam';

  const result: APIResult<{ response: { players: SteamPlayer[] } }> = useSteamUser(
    apiKey:  string | null,
    steamId: string | null,
  );

Parameters:
  apiKey   string | null   Steam Web API key. Pass null to disable.
  steamId  string | null   64-bit SteamID as a string. Pass null to disable.

Returns: APIResult<{ response: { players: SteamPlayer[] } }>
  data.response.players[0]  SteamPlayer   The requested player (array always has one entry)

SteamPlayer fields:
  steamid                  string
  personaname              string             Display name
  profileurl               string
  avatar                   string             32x32 avatar URL
  avatarmedium             string             64x64 avatar URL
  avatarfull               string             184x184 avatar URL
  personastate             number             Online state (0=offline, 1=online, 2=busy, 3=away, 4=snooze, 5=looking to trade, 6=looking to play)
  communityvisibilitystate number             1=private, 3=public
  lastlogoff               number | undefined Unix timestamp
  timecreated              number | undefined Account creation timestamp
  loccountrycode           string | undefined Two-letter country code
  locstatecode             string | undefined State/province code
  gameextrainfo            string | undefined Name of game currently being played
  gameid                   string | undefined App ID of game currently being played

--- useSteamOwnedGames ---

Fetch the full game library for a Steam user.

  import { useSteamOwnedGames } from '@reactjit/apis/steam';

  const result: APIResult<{ response: { game_count: number; games: SteamOwnedGame[] } }> = useSteamOwnedGames(
    apiKey:  string | null,
    steamId: string | null,
    opts?:   { includeAppInfo?: boolean },
  );

Parameters:
  apiKey                string | null   Steam API key. Pass null to disable.
  steamId               string | null   64-bit SteamID. Pass null to disable.
  opts.includeAppInfo   boolean         Include game name and icon hash in each entry. Defaults to true.

Returns: APIResult<{ response: { game_count: number; games: SteamOwnedGame[] } }>
  data.response.game_count  number
  data.response.games       SteamOwnedGame[]

SteamOwnedGame fields:
  appid                           number
  name                            string
  playtime_forever                number             Total playtime in minutes
  playtime_2weeks                 number | undefined Playtime in the last 2 weeks
  img_icon_url                    string             Icon hash — pass to steamGameIcon()
  has_community_visible_stats     boolean
  rtime_last_played               number             Unix timestamp of last session

--- useSteamRecentGames ---

Fetch the games a user has played in the last two weeks, sorted by recent playtime.

  import { useSteamRecentGames } from '@reactjit/apis/steam';

  const result: APIResult<{ response: { total_count: number; games: SteamRecentGame[] } }> = useSteamRecentGames(
    apiKey:  string | null,
    steamId: string | null,
    opts?:   { count?: number },
  );

Parameters:
  apiKey       string | null   Steam API key. Pass null to disable.
  steamId      string | null   64-bit SteamID. Pass null to disable.
  opts.count   number          Maximum number of games to return. Defaults to 10.

Returns: APIResult<{ response: { total_count: number; games: SteamRecentGame[] } }>
  data.response.total_count  number
  data.response.games        SteamRecentGame[]

SteamRecentGame fields:
  appid              number
  name               string
  playtime_2weeks    number   Minutes played in last 2 weeks
  playtime_forever   number   Total minutes played
  img_icon_url       string   Icon hash — pass to steamGameIcon()

--- useSteamFriends ---

Fetch the friend list for a Steam user. The profile must be public for this to succeed.

  import { useSteamFriends } from '@reactjit/apis/steam';

  const result: APIResult<{ friendslist: { friends: SteamFriend[] } }> = useSteamFriends(
    apiKey:  string | null,
    steamId: string | null,
  );

Parameters:
  apiKey   string | null
  steamId  string | null

Returns: APIResult<{ friendslist: { friends: SteamFriend[] } }>
  data.friendslist.friends  SteamFriend[]

SteamFriend fields:
  steamid       string
  relationship  string    Always 'friend' (only confirmed friends are returned)
  friend_since  number    Unix timestamp when the friendship was established

--- useSteamAchievements ---

Fetch the achievement status for a specific user and game combination.

  import { useSteamAchievements } from '@reactjit/apis/steam';

  const result: APIResult<{ playerstats: { achievements: SteamAchievement[] } }> = useSteamAchievements(
    apiKey:  string | null,
    steamId: string | null,
    appId:   number | null,
  );

Parameters:
  apiKey   string | null   Steam API key. Pass null to disable.
  steamId  string | null   64-bit SteamID. Pass null to disable.
  appId    number | null   Steam App ID (e.g. 570 for Dota 2). Pass null to disable.

Returns: APIResult<{ playerstats: { achievements: SteamAchievement[] } }>
  data.playerstats.achievements  SteamAchievement[]

SteamAchievement fields:
  apiname      string             Internal achievement identifier
  achieved     number             1 if unlocked, 0 if not
  unlocktime   number             Unix timestamp of unlock (0 if not achieved)
  name         string | undefined Human-readable name (included when the game schema is available)
  description  string | undefined Achievement description

--- useSteamAppDetails ---

Fetch store metadata for a game from the Steam Store API. No API key required.

  import { useSteamAppDetails } from '@reactjit/apis/steam';

  const result: APIResult<Record<string, { success: boolean; data: SteamAppDetails }>> = useSteamAppDetails(
    appId: number | null,
  );

Parameters:
  appId  number | null   Steam App ID. Pass null to disable.

Returns: APIResult<Record<string, { success: boolean; data: SteamAppDetails }>>
  data[appId].success   boolean
  data[appId].data      SteamAppDetails

SteamAppDetails fields:
  name                        string
  steam_appid                 number
  is_free                     boolean
  short_description           string
  header_image                string             Header image URL
  developers                  string[]
  publishers                  string[]
  price_overview              { final_formatted: string; discount_percent: number } | undefined
  metacritic                  { score: number; url: string } | undefined
  categories                  Array<{ description: string }>
  genres                      Array<{ description: string }>
  release_date                { coming_soon: boolean; date: string }

--- steamGameIcon (utility) ---

Construct a Steam CDN URL for a game's icon image.

  import { steamGameIcon } from '@reactjit/apis/steam';

  const url: string = steamGameIcon(appId: number, iconHash: string);

Parameters:
  appId      number   Steam App ID
  iconHash   string   The img_icon_url value from SteamOwnedGame or SteamRecentGame

Returns: string   Full JPEG URL

--- steamHeaderImage (utility) ---

Construct a Steam CDN URL for a game's 460x215 header image. No icon hash needed — only the App ID.

  import { steamHeaderImage } from '@reactjit/apis/steam';

  const url: string = steamHeaderImage(appId: number);

Parameters:
  appId  number   Steam App ID

Returns: string   Full JPEG URL for the store header image

=== EXAMPLES ===

-- Now playing widget --

Display what game the user is currently playing, or their last-seen status if offline.

  import { useSteamUser } from '@reactjit/apis/steam';
  import { Box, Text, Image } from '@reactjit/core';

  const API_KEY  = 'your-steam-api-key';
  const STEAM_ID = '76561197960435530';

  function NowPlaying() {
    const { data, loading } = useSteamUser(API_KEY, STEAM_ID);
    const player = data?.response.players[0];

    if (loading || !player) return <Text fontSize={14}>Loading...</Text>;

    const STATES = ['Offline', 'Online', 'Busy', 'Away', 'Snooze'];

    return (
      <Box flexDirection="row" gap={12} alignItems="center" padding={12}>
        <Image src={player.avatarmedium} width={64} height={64} />
        <Box gap={4}>
          <Text fontSize={15}>{player.personaname}</Text>
          {player.gameextrainfo ? (
            <Text fontSize={12}>Playing: {player.gameextrainfo}</Text>
          ) : (
            <Text fontSize={12}>{STATES[player.personastate] ?? 'Unknown'}</Text>
          )}
        </Box>
      </Box>
    );
  }

-- Recently played games list --

Show the last 5 games played with total and recent playtime, plus a header image.

  import { useSteamRecentGames, steamHeaderImage } from '@reactjit/apis/steam';
  import { Box, Text, Image } from '@reactjit/core';

  const API_KEY  = 'your-steam-api-key';
  const STEAM_ID = '76561197960435530';

  function RecentGames() {
    const { data, loading } = useSteamRecentGames(API_KEY, STEAM_ID, { count: 5 });
    const games = data?.response.games ?? [];

    if (loading && !data) return <Text fontSize={14}>Loading library...</Text>;

    return (
      <Box width="100%" gap={8} padding={12}>
        <Text fontSize={16}>Recently Played</Text>
        {games.map((game) => {
          const hours2w = (game.playtime_2weeks / 60).toFixed(1);
          const hoursAll = (game.playtime_forever / 60).toFixed(0);
          return (
            <Box key={game.appid} flexDirection="row" gap={10} alignItems="center">
              <Image src={steamHeaderImage(game.appid)} width={120} height={45} />
              <Box gap={3}>
                <Text fontSize={13}>{game.name}</Text>
                <Text fontSize={11}>{hours2w}h last 2 weeks / {hoursAll}h total</Text>
              </Box>
            </Box>
          );
        })}
      </Box>
    );
  }
