=== OVERVIEW ===
The YNAB hooks in @reactjit/apis connect to the YNAB REST API v1. Authentication requires a Personal Access Token generated at https://app.ynab.com/settings/developer.

All monetary values returned by the YNAB API are in milliunits (e.g., $12.34 is returned as 12340). Use the ynabAmount() utility to convert to dollars before displaying. Negative values represent outflows (spending); positive values represent inflows (income).

Most hooks accept an optional budgetId. When omitted or passed as null, they default to 'last-used', which resolves to the budget most recently opened in the YNAB app.

=== API / SYNTAX ===

--- useYNABBudgets ---

Fetch all budgets accessible with the token.

  import { useYNABBudgets } from '@reactjit/apis';

  const result: APIResult<{ data: { budgets: YNABBudget[] } }> = useYNABBudgets(
    token: string | null,
  );

Parameters:
  token  string | null  YNAB Personal Access Token

Returns: APIResult<{ data: { budgets: YNABBudget[] } }>

YNABBudget fields:
  id               string
  name             string
  last_modified_on string   ISO 8601 datetime of last change
  first_month      string   Earliest month in the budget (YYYY-MM-DD)
  last_month       string   Latest month in the budget (YYYY-MM-DD)
  currency_format  object
    iso_code           string   e.g. 'USD'
    decimal_digits     number
    decimal_separator  string

--- useYNABAccounts ---

Fetch all accounts in a budget.

  import { useYNABAccounts } from '@reactjit/apis';

  const result: APIResult<{ data: { accounts: YNABAccount[] } }> = useYNABAccounts(
    token: string | null,
    budgetId: string | null,
  );

Parameters:
  token     string | null  YNAB Personal Access Token
  budgetId  string | null  Budget ID, or null to use last-used

Returns: APIResult<{ data: { accounts: YNABAccount[] } }>

YNABAccount fields:
  id                  string
  name                string
  type                string    'checking', 'savings', 'creditCard', 'cash', etc.
  on_budget           boolean   Whether the account is on-budget
  closed              boolean
  balance             number    Current balance in milliunits
  cleared_balance     number    Cleared balance in milliunits
  uncleared_balance   number    Uncleared balance in milliunits
  transfer_payee_id   string    Payee ID for transfers to this account

--- useYNABCategories ---

Fetch all category groups and their categories for a budget month.

  import { useYNABCategories } from '@reactjit/apis';

  const result: APIResult<{ data: { category_groups: YNABCategoryGroup[] } }> = useYNABCategories(
    token: string | null,
    budgetId?: string | null,
  );

Parameters:
  token     string | null   YNAB Personal Access Token
  budgetId  string | null   Budget ID, or null/omitted to use last-used

Returns: APIResult<{ data: { category_groups: YNABCategoryGroup[] } }>

YNABCategoryGroup fields:
  id          string
  name        string
  hidden      boolean
  categories  YNABCategory[]

YNABCategory fields:
  id                        string
  name                      string
  category_group_id         string
  budgeted                  number   Amount budgeted this month in milliunits
  activity                  number   Amount spent this month in milliunits (negative = spending)
  balance                   number   Remaining balance in milliunits
  goal_type                 string | null   'TB', 'TBD', 'MF', 'NEED', or null
  goal_percentage_complete  number | null   0–100 when a goal exists

--- useYNABTransactions ---

Fetch transactions with optional date and type filtering.

  import { useYNABTransactions } from '@reactjit/apis';

  const result: APIResult<{ data: { transactions: YNABTransaction[] } }> = useYNABTransactions(
    token: string | null,
    opts?: { budgetId?: string; sinceDate?: string; type?: 'uncategorized' | 'unapproved' },
  );

Parameters:
  token  string | null   YNAB Personal Access Token
  opts   object | undefined
    budgetId   string    Budget ID (default: last-used)
    sinceDate  string    Only return transactions on or after this date (YYYY-MM-DD)
    type       string    'uncategorized' or 'unapproved' to filter

Returns: APIResult<{ data: { transactions: YNABTransaction[] } }>

YNABTransaction fields:
  id               string
  date             string              YYYY-MM-DD
  amount           number              Transaction amount in milliunits (negative = outflow)
  memo             string | null
  cleared          'cleared' | 'uncleared' | 'reconciled'
  approved         boolean
  payee_name       string | null
  category_name    string | null
  account_name     string
  subtransactions  Array<{ amount: number; memo: string | null; category_name: string | null }>

--- useYNABMonths ---

Fetch monthly summary data for a budget: income, budgeted, spending, and age of money.

  import { useYNABMonths } from '@reactjit/apis';

  const result: APIResult<{ data: { months: YNABMonth[] } }> = useYNABMonths(
    token: string | null,
    budgetId?: string | null,
  );

Parameters:
  token     string | null   YNAB Personal Access Token
  budgetId  string | null   Budget ID (default: last-used)

Returns: APIResult<{ data: { months: YNABMonth[] } }>

YNABMonth fields:
  month             string   YYYY-MM-DD (first day of the month)
  income            number   Total income in milliunits
  budgeted          number   Total budgeted in milliunits
  activity          number   Total spending in milliunits (negative)
  to_be_budgeted    number   Unallocated funds in milliunits
  age_of_money      number | null   Average age of money in days

--- ynabAmount (utility) ---

Convert YNAB milliunits to a dollar (or local currency) amount. Not a hook — call it inline.

  import { ynabAmount } from '@reactjit/apis';

  const dollars: number = ynabAmount(milliunits: number);

Parameters:
  milliunits  number  YNAB milliunit value (e.g. 12340)

Returns: number — human-readable amount (e.g. 12.34)

=== EXAMPLES ===

-- Category budget progress bars --

Show each category's spending against its monthly budget:

  import { useYNABCategories, ynabAmount } from '@reactjit/apis';
  import { Box, Text } from '@reactjit/core';

  const TOKEN = process.env.YNAB_TOKEN;

  function BudgetProgress() {
    const { data, loading } = useYNABCategories(TOKEN);

    if (loading) return <Text fontSize={13}>Loading budget...</Text>;

    const groups = data?.data?.category_groups ?? [];
    // Skip hidden internal groups like 'Internal Master Category'
    const visible = groups.filter((g) => !g.hidden && g.name !== 'Internal Master Category');

    return (
      <Box gap={16} width="100%">
        {visible.map((group) => (
          <Box key={group.id} gap={6}>
            <Text fontSize={13}>{group.name}</Text>
            {group.categories.map((cat) => {
              const budgeted = ynabAmount(cat.budgeted);
              // activity is negative for spending
              const spent = ynabAmount(Math.abs(cat.activity));
              const pct = budgeted > 0 ? Math.min(spent / budgeted, 1) : 0;
              const overBudget = spent > budgeted;

              return (
                <Box key={cat.id} gap={4}>
                  <Box flexDirection="row" justifyContent="space-between" width="100%">
                    <Text fontSize={12}>{cat.name}</Text>
                    <Text fontSize={12}>
                      ${spent.toFixed(2)} / ${budgeted.toFixed(2)}
                    </Text>
                  </Box>
                  <Box width="100%" height={6} backgroundColor="#333" borderRadius={3}>
                    <Box
                      width={`${Math.round(pct * 100)}%`}
                      height={6}
                      backgroundColor={overBudget ? '#D32F2F' : '#4CAF50'}
                      borderRadius={3}
                    />
                  </Box>
                </Box>
              );
            })}
          </Box>
        ))}
      </Box>
    );
  }

-- Monthly summary widget --

Show this month's income, spending, and to-be-budgeted:

  import { useYNABMonths, ynabAmount } from '@reactjit/apis';
  import { Box, Text } from '@reactjit/core';

  const TOKEN = process.env.YNAB_TOKEN;

  function MonthlySummary() {
    const { data, loading } = useYNABMonths(TOKEN);

    if (loading) return <Text fontSize={13}>Loading...</Text>;

    const months = data?.data?.months ?? [];
    // Most recent month is last in the array
    const current = months[months.length - 1];
    if (!current) return null;

    const income = ynabAmount(current.income);
    const spent = ynabAmount(Math.abs(current.activity));
    const tbb = ynabAmount(current.to_be_budgeted);

    return (
      <Box gap={8} padding={16} backgroundColor="#0d1117">
        <Text fontSize={14}>{current.month.slice(0, 7)}</Text>
        <Box flexDirection="row" gap={24}>
          <Box>
            <Text fontSize={11}>Income</Text>
            <Text fontSize={16}>${income.toFixed(2)}</Text>
          </Box>
          <Box>
            <Text fontSize={11}>Spent</Text>
            <Text fontSize={16}>${spent.toFixed(2)}</Text>
          </Box>
          <Box>
            <Text fontSize={11}>Left to Budget</Text>
            <Text fontSize={16}>${tbb.toFixed(2)}</Text>
          </Box>
        </Box>
      </Box>
    );
  }
