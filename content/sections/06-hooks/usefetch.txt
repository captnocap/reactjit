=== OVERVIEW ===
useFetch makes HTTP requests through the Lua runtime. In Love2D it uses the fetch() polyfill which routes to LuaSocket for HTTP or love.filesystem for local files. Returns standard loading/error/data state.

Pass `null` as the URL to disable fetching. The hook re-fetches when the URL changes. Options follow the standard RequestInit interface (method, headers, body).

=== API / SYNTAX ===
```tsx
import { useFetch } from '@reactjit/core';
```

Signature:
```tsx
function useFetch<T = any>(
  url: string | null,
  options?: RequestInit
): { data: T | null; error: Error | null; loading: boolean }
```

Parameters:
| Param   | Type            | Default   | Description                          |
|---------|-----------------|-----------|--------------------------------------|
| url     | string \| null  | required  | URL to fetch, or null to disable     |
| options | RequestInit     | undefined | Standard fetch options (method, etc) |

Returns:
| Field   | Type          | Description                              |
|---------|---------------|------------------------------------------|
| data    | T \| null     | Parsed JSON response, null until loaded  |
| error   | Error \| null | Error if the request failed              |
| loading | boolean       | True while the request is in flight      |

=== EXAMPLES ===
Example 1: Fetch JSON data
---
```tsx
import { Box, Text } from '@reactjit/core';
import { useFetch } from '@reactjit/core';
import { useThemeColors } from '@reactjit/theme';

interface User { id: number; name: string; }

export function UserList() {
  const colors = useThemeColors();
  const { data, loading, error } = useFetch<User[]>('https://api.example.com/users');

  if (loading) return <Text style={{ color: colors.muted }}>Loading...</Text>;
  if (error) return <Text style={{ color: colors.primary }}>{`Error: ${error.message}`}</Text>;

  return (
    <Box style={{ gap: 8 }}>
      {data?.map(user => (
        <Text key={user.id} style={{ color: colors.text }}>{user.name}</Text>
      ))}
    </Box>
  );
}
```

Example 2: Conditional fetching
---
```tsx
import { useState } from 'react';
import { Box, Text, Pressable } from '@reactjit/core';
import { useFetch } from '@reactjit/core';
import { useThemeColors } from '@reactjit/theme';

export function ConditionalFetch() {
  const colors = useThemeColors();
  const [userId, setUserId] = useState<number | null>(null);
  const { data, loading } = useFetch(userId ? `https://api.example.com/users/${userId}` : null);

  return (
    <Box style={{ gap: 8 }}>
      <Pressable onPress={() => setUserId(1)}>
        <Text style={{ color: colors.primary }}>Load User 1</Text>
      </Pressable>
      {loading && <Text style={{ color: colors.muted }}>Loading...</Text>}
      {data && <Text style={{ color: colors.text }}>{`Name: ${data.name}`}</Text>}
    </Box>
  );
}
```

Platforms: love2d

Example 3: Load a local JSON file
---
```tsx
import { useFetch } from '@reactjit/core';

interface Config { theme: string; fontSize: number; }

export function useConfig() {
  return useFetch<Config>('data/config.json');
}
```

=== CRITICAL RULES ===
- The response is always parsed as JSON. If you need raw text, use bridge.rpc or fetch directly.
- Passing null as the URL resets data, error, and loading to their initial states.
- The hook cancels in-flight requests on unmount or URL change to prevent stale updates.
- Options are captured by ref -- changing options alone does not trigger a re-fetch. Only URL changes trigger re-fetch.
