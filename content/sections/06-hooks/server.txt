=== OVERVIEW ===
@reactjit/server provides React hooks that start and manage a non-blocking
HTTP server running inside Love2D via lua/httpserver.lua. Static files (media,
assets) are served entirely from the Lua side with zero bridge overhead. Dynamic
API routes forward requests across the QuickJS bridge to React handlers that
return response objects. useLibrary adds on-demand filesystem indexing so you
can build searchable media servers with a single call.

Import: import { useServer, useStaticServer, useLibrary } from '@reactjit/server';

=== API / SYNTAX ===

---- useServer(config) -------------------------------------------------------

Start an HTTP server. The server is bound on mount and torn down on unmount.
Changing port or host restarts the server; changing routes or static entries
does not restart it (route table updates are live).

  useServer(config: ServerConfig | null): UseServerResult

  Arguments:
    config  ServerConfig | null   Pass null to disable the server without
                                  unmounting the hook.

  ServerConfig:
    port     number           Required. TCP port to listen on.
    host?    string           Bind address. Default: '0.0.0.0' (all interfaces).
    routes?  Route[]          Dynamic API routes handled in React.
    static?  StaticRoute[]    Static directories served from Lua.

  Route:
    method?   string         HTTP method filter ('GET', 'POST', ...).
                             Omit to match any method.
    path      string         Path pattern. Supports :param and * wildcards.
    handler   RouteHandler   Function called when a request matches.

  RouteHandler:
    (req: HttpRequest) => HttpResponse | Promise<HttpResponse>

  HttpRequest:
    clientId  number                   Internal client socket id.
    serverId  string                   Internal server instance id.
    method    string                   HTTP method (uppercase).
    path      string                   Matched path with params resolved.
    rawPath   string                   Exact path as received from the client.
    query     Record<string, string>   Parsed query string parameters.
    headers   Record<string, string>   Request headers (lowercase keys).
    body      string                   Raw request body string.
    params    Record<string, string>   Named URL parameters from the route pattern.
    route     string                   The route pattern that matched.

  HttpResponse:
    status    number                     HTTP status code.
    headers?  Record<string, string>     Response headers.
    body?     string                     Response body string.

  StaticRoute:
    path   string   URL prefix to match (e.g. '/media', '/').
    root   string   Absolute OS filesystem path (e.g. '/home/user/music').

  UseServerResult:
    ready     boolean         True once the socket is bound and listening.
    port      number | null   Listening port, or null if not ready.
    requests  HttpRequest[]   Recent requests, most recent first. Capped at 50.
    close     () => void      Manually shut down the server.

---- useStaticServer(port, root, host?) -------------------------------------

One-liner for serving a single directory over HTTP. A convenience wrapper
around useServer with a single static route at path '/'.

  useStaticServer(
    port:   number | null,
    root:   string,
    host?:  string,
  ): UseServerResult

  Arguments:
    port   number | null   Port to listen on. Pass null to disable.
    root   string          Absolute path to the directory to serve.
    host   string          Bind address. Default: '0.0.0.0'.

  Returns the same UseServerResult as useServer.

---- useLibrary(port, directories, host?) -----------------------------------

Index one or more directories, start a server that serves the files directly,
and expose a searchable JSON index endpoint. Files are served by Lua (zero
bridge overhead). The React side receives a full in-memory file index with
category breakdowns and per-directory stats.

  useLibrary(
    port:         number | null,
    directories:  string[],
    host?:        string,
  ): UseLibraryResult

  Arguments:
    port         number | null   Port. Pass null to disable.
    directories  string[]        Absolute OS paths to index and serve.
                                 Directory name (last path segment) becomes
                                 the URL namespace.
    host         string          Bind address. Default: '0.0.0.0'.

  UseLibraryResult:
    ready        boolean              True once server is up and index is built.
    port         number | null        Listening port, or null if not ready.
    files        IndexedFile[]        Full flat file index.
    stats        LibraryIndex.stats   Count totals by category.
    directories  DirectoryStats[]     Per-directory name/path/count/size.
    reindex      () => Promise<void>  Re-scan all directories and rebuild index.
    close        () => void           Shut down the server.

  IndexedFile:
    name      string        Filename with extension (e.g. 'song.mp3').
    path      string        Absolute OS path.
    relPath   string        Path relative to the indexed directory root.
    size      number        File size in bytes.
    modified  number        Last modified epoch seconds.
    ext       string | null Lowercase file extension (e.g. 'mp3'), or null.
    category  FileCategory  'audio' | 'video' | 'image' | 'document' | 'other'
    dir       string        Name of the parent directory (matches URL namespace).

  DirectoryStats:
    name   string   Directory name (last path segment).
    path   string   Absolute OS path.
    count  number   Total files indexed in this directory.
    size   number   Total bytes of all files.

  Auto-served endpoints (directory name derived from last path segment):
    GET /files/<dir>/<relPath>   Serve actual file (Lua, zero bridge).
    GET /api/library             Full JSON index.
    GET /api/library?type=audio  Filter by category (audio/video/image/...).
    GET /api/library?dir=music   Filter by directory name.
    GET /api/library?q=<term>    Search by filename (case-insensitive).

=== EXAMPLES ===

---- Basic API server with JSON endpoints -----------------------------------

import { useServer } from '@reactjit/server';

function App() {
  const [status, setStatus] = React.useState('ok');

  const server = useServer({
    port: 8080,
    routes: [
      {
        method: 'GET',
        path: '/api/status',
        handler: () => ({
          status: 200,
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ status }),
        }),
      },
      {
        method: 'POST',
        path: '/api/action',
        handler: async (req) => {
          const data = JSON.parse(req.body);
          // handle action...
          return { status: 204 };
        },
      },
    ],
  });

  return (
    <Box width="100%" height="100%">
      <Text fontSize={14}>
        {server.ready ? `Listening on :${server.port}` : 'Starting...'}
      </Text>
    </Box>
  );
}

    </Box>
  );
}

---- Mixed server: static files + dynamic API ------------------------------

import { useServer } from '@reactjit/server';

function MediaServer() {
  const [playlist, setPlaylist] = React.useState<string[]>([]);

  const server = useServer({
    port: 8888,
    static: [
      { path: '/music',  root: '/home/user/music' },
      { path: '/photos', root: '/home/user/photos' },
    ],
    routes: [
      {
        method: 'GET',
        path: '/api/playlist',
        handler: () => ({
          status: 200,
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(playlist),
        }),
      },
      {
        method: 'POST',
        path: '/api/playlist/add',
        handler: async (req) => {
          const { track } = JSON.parse(req.body);
          setPlaylist((prev) => [...prev, track]);
          return { status: 200, body: 'OK' };
        },
      },
    ],
  });

  return (
    <Text fontSize={14}>
      {server.ready ? `Server on :${server.port}` : 'Starting...'}
    </Text>
  );
}

---- URL parameters in dynamic routes ---------------------------------------

import { useServer } from '@reactjit/server';

const server = useServer({
  port: 9000,
  routes: [
    {
      method: 'GET',
      path: '/api/users/:id',
      handler: async (req) => {
        const { id } = req.params;
        const user = await lookupUser(id);
        if (!user) return { status: 404, body: 'Not found' };
        return {
          status: 200,
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(user),
        };
      },
    },
  ],
});

---- Query string parameters ------------------------------------------------

import { useServer } from '@reactjit/server';

const server = useServer({
  port: 9001,
  routes: [
    {
      path: '/api/search',
      handler: (req) => {
        const query = req.query.q ?? '';
        const limit = parseInt(req.query.limit ?? '20', 10);
        const results = search(query, limit);
        return {
          status: 200,
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(results),
        };
      },
    },
  ],
});

// Incoming: GET /api/search?q=beethoven&limit=10
// req.query === { q: 'beethoven', limit: '10' }

---- Media library with index and file serving --------------------------------

import { useLibrary } from '@reactjit/server';
import { Box, Text } from '@reactjit/core';

function LibraryApp() {
  const library = useLibrary(8080, [
    '/home/user/music',
    '/home/user/movies',
    '/home/user/photos',
  ]);

  if (!library.ready) return <Text fontSize={14}>Indexing...</Text>;

  return (
    <Box width="100%" height="100%" flexDirection="column">
      <Text fontSize={18}>
        {library.stats.total} files indexed
      </Text>
      <Text fontSize={14}>
        Audio: {library.stats.audio ?? 0} |
        Video: {library.stats.video ?? 0} |
        Images: {library.stats.image ?? 0}
      </Text>
      {library.directories.map(dir => (
        <Text key={dir.path} fontSize={12}>
          {dir.name}: {dir.count} files
        </Text>
      ))}
    </Box>
  );
}

// A client can now fetch:
//   http://device:8080/files/music/Artist/Album/track.flac
//   http://device:8080/files/movies/film.mkv
//   http://device:8080/api/library?type=audio
//   http://device:8080/api/library?q=beethoven

---- Reindex on demand -------------------------------------------------------

import { useLibrary } from '@reactjit/server';

function LibraryWithRefresh() {
  const library = useLibrary(8080, ['/home/user/media']);

  return (
    <Box>
      <Text fontSize={14}>{library.stats.total} files</Text>
      <Pressable onPress={() => library.reindex()}>
        <Text fontSize={14}>Refresh Library</Text>
      </Pressable>
    </Box>
  );
}

---- Show recent request log ------------------------------------------------

import { useServer } from '@reactjit/server';

function ServerLog() {
  const server = useServer({ port: 8080, routes: [] });

  return (
    <Box width="100%" height="100%">
      <Text fontSize={14}>Recent requests (up to 50):</Text>
      {server.requests.map((req, i) => (
        <Text key={i} fontSize={12}>
          {req.method} {req.rawPath}
        </Text>
      ))}
    </Box>
  );
}

---- Disable server conditionally -------------------------------------------

import { useStaticServer } from '@reactjit/server';

function App() {
  const [enabled, setEnabled] = React.useState(false);
  const server = useStaticServer(enabled ? 8080 : null, '/home/user/files');

  return (
    <Pressable onPress={() => setEnabled(e => !e)}>
      <Text fontSize={16}>{enabled ? 'Stop server' : 'Start server'}</Text>
    </Pressable>
  );
}
