=== OVERVIEW ===
ReactJIT includes a lightweight spell checker (lua/spellcheck.lua) powered by a SQLite dictionary with 50,000+ frequency-ranked English words. It checks words against the dictionary and generates suggestions using edit-distance-1 candidates (insertions, deletions, substitutions, transpositions). Designed for real-time use in TextInput and TextEditor components, accessible from React via the RPC bridge. Requires a dictionary.db file in data/ and libsqlite3.so availability.

=== API / SYNTAX ===

**Lua API (lua/spellcheck.lua)**

```lua
local spellcheck = require("lua.spellcheck")

-- Initialize (loads dictionary.db)
spellcheck.init()                        -- defaults to data/dictionary.db, lang="en"
spellcheck.init({ lang = "en" })         -- specify language

-- Check availability
if not spellcheck.available then
  -- SQLite or dictionary.db not found
end

-- Check single word (case-insensitive)
spellcheck.check("hello")                -- true
spellcheck.check("helo")                 -- false

-- Check text, get error positions
local errors = spellcheck.checkText("I hav a speling eror")
-- Returns: {
--   { word = "hav", start = 3, stop = 5 },
--   { word = "speling", start = 9, stop = 15 },
--   { word = "eror", start = 17, stop = 20 }
-- }

-- Get suggestions (frequency-ranked, edit-distance-1)
local suggestions = spellcheck.suggest("helo")      -- {"hello", "help", "hero", ...}
local top3 = spellcheck.suggest("helo", 3)          -- limit results

-- Language management
spellcheck.setLang("fr")                 -- switch language (if dictionary has French words)
local lang = spellcheck.getLang()        -- "en"

-- Add custom word
spellcheck.addWord("reactjit")
spellcheck.addWord("reactjit", 5000)   -- with frequency

-- Remove word
spellcheck.removeWord("reactjit")
```

**React API (via RPC bridge)**

Available RPC handlers (registered in lua/init.lua):

```tsx
import { useBridge } from '@reactjit/core';

function SpellCheckComponent() {
  const bridge = useBridge();

  // Check single word
  const isCorrect = await bridge.rpc('spell:check', { word: 'hello' });

  // Check text, get errors
  const errors = await bridge.rpc('spell:checkText', {
    text: 'I hav a speling eror'
  });
  // Returns: [{ word, start, stop }, ...]

  // Get suggestions
  const suggestions = await bridge.rpc('spell:suggest', {
    word: 'helo',
    limit: 5
  });
  // Returns: ["hello", "help", "hero", ...]

  // Set language
  await bridge.rpc('spell:setLang', { lang: 'en' });
}
```

**Dictionary Schema (SQLite)**

```sql
CREATE TABLE words (
  word TEXT NOT NULL,
  freq INTEGER NOT NULL DEFAULT 0,  -- frequency ranking (higher = more common)
  lang TEXT NOT NULL DEFAULT 'en',
  PRIMARY KEY (word, lang)
);

CREATE INDEX idx_words_lang ON words(lang);
```

Example 2: React TextInput with Real-Time Spell Check
---
import React, { useState, useEffect, useCallback } from 'react';
import { Box, Text, TextInput, useBridge } from '@reactjit/core';

function SpellCheckedInput() {
  const bridge = useBridge();
  const [text, setText] = useState('I hav a speling eror');
  const [errors, setErrors] = useState<Array<{ word: string; start: number; stop: number }>>([]);
  const [available, setAvailable] = useState(true);

  // Debounced spell check
  useEffect(() => {
    const timer = setTimeout(async () => {
      try {
        const result = await bridge.rpc('spell:checkText', { text });
        setErrors(result || []);
      } catch {
        setAvailable(false);
      }
    }, 300);

    return () => clearTimeout(timer);
  }, [text, bridge]);

  if (!available) {
    return <Text fontSize={14} color="#888">Spell check unavailable</Text>;
  }

  return (
    <Box style={{ gap: 12 }}>
      <TextInput
        value={text}
        onChangeText={setText}
        multiline
        style={{ fontSize: 14, color: '#fff', minHeight: 60 }}
      />
      <Box style={{ gap: 4 }}>
        {errors.length === 0 ? (
          <Text fontSize={12} color="#4ade80">No spelling errors</Text>
        ) : (
          <Text fontSize={12} color="#f87171">
            {errors.length} error{errors.length > 1 ? 's' : ''}: {errors.map(e => e.word).join(', ')}
          </Text>
        )}
      </Box>
    </Box>
  );
}

Example 3: Spell Check with Suggestions and Corrections
---
import React, { useState, useEffect, useCallback } from 'react';
import { Box, Text, Pressable, TextInput, useBridge } from '@reactjit/core';

function SpellCheckWithSuggestions() {
  const bridge = useBridge();
  const [input, setInput] = useState('I hav a speling eror in this sentance');
  const [errors, setErrors] = useState<any[]>([]);
  const [suggestions, setSuggestions] = useState<Record<string, string[]>>({});

  // Check text (debounced)
  useEffect(() => {
    const timer = setTimeout(async () => {
      const errs = await bridge.rpc('spell:checkText', { text: input });
      setErrors(errs || []);

      // Fetch suggestions for each error
      const sugs: Record<string, string[]> = {};
      for (const err of (errs || [])) {
        const s = await bridge.rpc('spell:suggest', { word: err.word, limit: 4 });
        sugs[err.word] = s || [];
      }
      setSuggestions(sugs);
    }, 300);

    return () => clearTimeout(timer);
  }, [input, bridge]);

  // Apply suggestion
  const applySuggestion = (misspelled: string, replacement: string) => {
    const regex = new RegExp(`\\b${misspelled}\\b`, 'i');
    setInput(prev => prev.replace(regex, replacement));
  };

  return (
    <Box style={{ gap: 16 }}>
      <TextInput
        value={input}
        onChangeText={setInput}
        multiline
        style={{ fontSize: 14, minHeight: 60 }}
      />

      {errors.map((err, i) => (
        <Box key={i} style={{ gap: 4 }}>
          <Text fontSize={13} color="#f87171">{err.word}</Text>
          <Box style={{ flexDirection: 'row', gap: 6, flexWrap: 'wrap' }}>
            {(suggestions[err.word] || []).map((sug, j) => (
              <Pressable
                key={j}
                onPress={() => applySuggestion(err.word, sug)}
                style={{
                  backgroundColor: '#1e293b',
                  paddingTop: 4,
                  paddingBottom: 4,
                  paddingLeft: 8,
                  paddingRight: 8,
                  borderRadius: 4,
                }}
              >
                <Text fontSize={12} color="#94a3b8">{sug}</Text>
              </Pressable>
            ))}
          </Box>
        </Box>
      ))}
    </Box>
  );
}

Example 4: Custom Dictionary Management
---
local spellcheck = require("lua.spellcheck")
spellcheck.init()

-- Add technical terms to dictionary
local function addTechnicalTerms()
  local terms = {
    { word = "reactjit", freq = 1000 },
    { word = "lua", freq = 2000 },
    { word = "opengl", freq = 1500 },
    { word = "vscode", freq = 1200 },
  }

  for _, term in ipairs(terms) do
    spellcheck.addWord(term.word, term.freq)
  end
end

-- Remove outdated words
local function cleanupDictionary()
  local deprecated = { "web2.0", "flashplayer", "myspace" }
  for _, word in ipairs(deprecated) do
    spellcheck.removeWord(word)
  end
end

addTechnicalTerms()
cleanupDictionary()

-- Now "reactjit" passes spell check
print(spellcheck.check("reactjit"))  -- true

=== CRITICAL RULES ===
- **Requires dictionary.db** — spell check silently disables if `data/dictionary.db` is not found. Check `spellcheck.available` before use.
- **Requires libsqlite3.so** — depends on the SQLite module. If SQLite is unavailable, spell check is unavailable. Always check `available` flag.
- **Case-insensitive checking** — all words are normalized to lowercase before dictionary lookup. "Hello", "HELLO", "hello" all check against "hello".
- **Skips numbers and URLs** — words starting with digits, containing `/`, or matching `http(s)://` are considered correct automatically.
- **Single-character words are ignored** — "a", "I" are always correct without dictionary lookup.
- **Edit-distance-1 only** — suggestions are limited to single-character edits (insert, delete, substitute, transpose). Multi-character typos like "hlelo" may have poor suggestions.
- **Frequency-ranked results** — suggestions are sorted by word frequency (most common words first). Higher frequency = more likely correct.
- **Debounce in React** — `checkText` performs a full scan + suggestion fetch. Debounce user input (300ms+) to avoid excessive RPC calls.
- **Apostrophes are stripped** — leading/trailing apostrophes are removed before checking ("don't" becomes "don't", but outer quotes are stripped).
- **Dictionary path is OS filesystem** — SQLite uses absolute paths or process CWD, NOT Love2D's save directory. The framework resolves `data/dictionary.db` relative to `love.filesystem.getSource()`.
- **addWord/removeWord persist in DB** — changes are written to the SQLite database immediately and survive app restarts.
- **No grammar checking** — this is a dictionary-based spell checker, not a grammar checker. "I are happy" has no spelling errors.
- **Language switching requires DB support** — `setLang()` changes the query filter, but the dictionary.db must contain words for that language. The default dictionary is English only.
