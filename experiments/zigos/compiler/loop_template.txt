    const main_window_id = c.SDL_GetWindowID(window);
    var running = true;

    while (running) {
        var event: c.SDL_Event = undefined;
        while (c.SDL_PollEvent(&event) != 0) {
            const evt_window_id: u32 = switch (event.type) {
                c.SDL_WINDOWEVENT => event.window.windowID,
                c.SDL_MOUSEMOTION => event.motion.windowID,
                c.SDL_MOUSEBUTTONDOWN, c.SDL_MOUSEBUTTONUP => event.button.windowID,
                c.SDL_MOUSEWHEEL => event.wheel.windowID,
                c.SDL_KEYDOWN, c.SDL_KEYUP => event.key.windowID,
                else => main_window_id,
            };
            if (evt_window_id != main_window_id) {
                if (win_mgr.findByWindowId(evt_window_id)) |win_idx| {
                    switch (event.type) {
                        c.SDL_WINDOWEVENT => {
                            if (event.window.event == c.SDL_WINDOWEVENT_CLOSE) { win_mgr.handleClose(win_idx); }
                            else if (event.window.event == c.SDL_WINDOWEVENT_SIZE_CHANGED) { win_mgr.handleResize(win_idx, event.window.data1, event.window.data2); }
                        },
                        c.SDL_MOUSEMOTION => { win_mgr.handleMouseMotion(win_idx, @floatFromInt(event.motion.x), @floatFromInt(event.motion.y)); },
                        c.SDL_MOUSEBUTTONDOWN => { win_mgr.handleClick(win_idx, @floatFromInt(event.button.x), @floatFromInt(event.button.y)); },
                        c.SDL_MOUSEWHEEL => {
                            var mx_i: c_int = undefined; var my_i: c_int = undefined;
                            _ = c.SDL_GetMouseState(&mx_i, &my_i);
                            win_mgr.handleWheel(win_idx, @floatFromInt(mx_i), @floatFromInt(my_i), @floatFromInt(event.wheel.y));
                        },
                        else => {},
                    }
                    continue;
                }
            }
            switch (event.type) {
                c.SDL_QUIT => running = false,
                c.SDL_WINDOWEVENT => {
                    if (event.window.event == c.SDL_WINDOWEVENT_SIZE_CHANGED) {
                        win_w = @floatFromInt(event.window.data1);
                        win_h = @floatFromInt(event.window.data2);
                        gpu.resize(@intCast(event.window.data1), @intCast(event.window.data2));
                        geometry.save(window);
                    }
                    if (event.window.event == c.SDL_WINDOWEVENT_MOVED) {
                        geometry.save(window);
                    }
                },
                c.SDL_TEXTINPUT => {
                    _onTextInput(@ptrCast(&event.text.text));
                },
                c.SDL_KEYDOWN => {
                    const ctrl = (event.key.keysym.mod & c.KMOD_CTRL) != 0;
                    // Route to focused input first
                    if (input_mod.getFocusedId() != null) {
                        if (ctrl and input_mod.handleCtrlKey(event.key.keysym.sym)) {
                            // Ctrl+A/C/X/V/Z handled
                        } else if (event.key.keysym.sym == c.SDLK_ESCAPE) { input_mod.unfocus(); }
                        else { _ = input_mod.handleKey(event.key.keysym.sym); }
                    } else if (ctrl and event.key.keysym.sym == c.SDLK_a) {
                        // Ctrl+A — select all text on the page
                        sel_all = true;
                        sel_node = null;
                        sel_start = 0;
                        sel_end = 0;
                    } else if (ctrl and event.key.keysym.sym == c.SDLK_c) {
                        // Ctrl+C — copy selected text to clipboard
                        var clip_buf: [16384]u8 = undefined;
                        var clip_len: usize = 0;
                        if (sel_all) {
                            clip_len = collectAllText(&root, &clip_buf, 0);
                        } else if (sel_node != null and sel_end_node != null) {
                            var copy_state: u8 = 0;
                            clip_len = collectSelectedText(&root, &clip_buf, 0, &copy_state);
                        }
                        if (clip_len > 0 and clip_len < clip_buf.len) {
                            clip_buf[clip_len] = 0;
                            _ = c.SDL_SetClipboardText(@ptrCast(&clip_buf));
                        }
                    } else if (event.key.keysym.sym == c.SDLK_F12) { inspector.toggle(); }
                    else if (event.key.keysym.sym == c.SDLK_ESCAPE) {
                        // Escape dismisses overlays first, then inspector, then quits
                        var any_overlay_visible = false;
                        var ov_i: usize = 0;
                        while (ov_i < overlay_mod.MAX_OVERLAYS) : (ov_i += 1) {
                            if (overlay_mod.isVisible(ov_i)) { any_overlay_visible = true; break; }
                        }
                        if (any_overlay_visible) {
                            overlay_mod.hideAll();
                        } else if (inspector.isEnabled()) {
                            if (inspector.getSelectedNode() != null) {
                                inspector.clearSelection();
                            } else {
                                inspector.toggle();
                            }
                        } else { running = false; }
                    }
                    else { const sdl_mod = event.key.keysym.mod; const mods: u16 = @intCast(sdl_mod & 0xFFFF); _onKeyDown(event.key.keysym.sym, mods); if (hovered_node) |node| { if (node.handlers.on_key) |handler| handler(event.key.keysym.sym, mods); } }
                },
                c.SDL_MOUSEMOTION => {
                    const mx: f32 = @floatFromInt(event.motion.x); const my: f32 = @floatFromInt(event.motion.y);
                    inspector.mouseMoved(mx, my);
                    // Check overlay hover first, then app tree
                    const overlay_hover = overlay_mod.hitTestOverlays(mx, my);
                    const prev = hovered_node;
                    hovered_node = if (overlay_hover != null) overlay_hover else events.hitTest(&root, mx, my);
                    if (prev != hovered_node) { if (prev) |p| { if (p.handlers.on_hover_exit) |h| h(); } if (hovered_node) |n| { if (n.handlers.on_hover_enter) |h| h(); } }
                    // Text selection drag — may cross into different text nodes
                    if (sel_dragging) {
                        if (events.hitTestText(&root, mx, my)) |tnode| {
                            if (tnode.text) |txt| {
                                const pad_l = tnode.style.padLeft();
                                const pad_r = tnode.style.padRight();
                                const pad_t = tnode.style.padTop();
                                const local_x = mx - tnode.computed.x - pad_l;
                                const local_y = my - tnode.computed.y - pad_t;
                                const text_max_w = tnode.computed.w - pad_l - pad_r;
                                sel_end_node = tnode;
                                sel_end = g_text_engine.?.hitTestWrappedAligned(txt, local_x, local_y, tnode.font_size, text_max_w, tnode.style.text_align);
                            }
                        }
                    }
                },
                c.SDL_MOUSEBUTTONDOWN => {
                    const mx: f32 = @floatFromInt(event.button.x); const my: f32 = @floatFromInt(event.button.y);
                    if (inspector.handleClick(&root, mx, my)) { continue; }
                    // Right-click dispatch
                    if (event.button.button == c.SDL_BUTTON_RIGHT) {
                        if (events.hitTest(&root, mx, my)) |node| {
                            if (node.handlers.on_right_click) |handler| handler(mx, my);
                        }
                        continue;
                    }
                    // Overlay click dispatch — runs before app hit testing
                    if (overlay_mod.handleClick(mx, my)) { continue; }
                    sel_all = false; // Any click clears select-all
                    // Handler-based hit test (buttons, inputs, etc.)
                    if (events.hitTest(&root, mx, my)) |node| {
                        if (node.input_id) |iid| {
                            input_mod.focus(iid);
                            sel_node = null;
                        } else {
                            input_mod.unfocus();
                            if (node.handlers.on_press) |handler| handler();
                        }
                    } else {
                        input_mod.unfocus();
                    }
                    // Text selection — only if no pressable/input was hit
                    const pressed_handler = if (events.hitTest(&root, mx, my)) |n| (n.handlers.on_press != null or n.input_id != null) else false;
                    if (!pressed_handler) {
                    if (events.hitTestText(&root, mx, my)) |tnode| {
                        if (tnode.text) |txt| {
                            const pad_l = tnode.style.padLeft();
                            const pad_r = tnode.style.padRight();
                            const pad_t = tnode.style.padTop();
                            const local_x = mx - tnode.computed.x - pad_l;
                            const local_y = my - tnode.computed.y - pad_t;
                            const text_max_w = tnode.computed.w - pad_l - pad_r;
                            const now = c.SDL_GetTicks();
                            const byte_idx = g_text_engine.?.hitTestWrappedAligned(txt, local_x, local_y, tnode.font_size, text_max_w, tnode.style.text_align);
                            if (sel_node == tnode and now -% sel_last_click < 400) {
                                sel_click_count += 1;
                            } else {
                                sel_click_count = 1;
                            }
                            sel_last_click = now;
                            if (sel_click_count >= 3) {
                                sel_node = tnode;
                                sel_end_node = tnode;
                                sel_start = 0;
                                sel_end = txt.len;
                                sel_anchor = 0;
                                sel_dragging = false;
                            } else if (sel_click_count == 2) {
                                const wb = TextEngine.wordBoundsAt(txt, byte_idx);
                                sel_node = tnode;
                                sel_end_node = tnode;
                                sel_start = wb.start;
                                sel_end = wb.end;
                                sel_anchor = wb.start;
                                sel_dragging = false;
                            } else {
                                sel_node = tnode;
                                sel_end_node = tnode;
                                sel_anchor = byte_idx;
                                sel_start = byte_idx;
                                sel_end = byte_idx;
                                sel_dragging = true;
                            }
                        }
                    } else {
                        sel_node = null;
                        sel_end_node = null;
                        sel_start = 0;
                        sel_end = 0;
                    }
                    }
                },
                c.SDL_MOUSEBUTTONUP => {
                    sel_dragging = false;
                    inspector.mouseReleased();
                },
                c.SDL_MOUSEWHEEL => {
                    var mx_i: c_int = undefined; var my_i: c_int = undefined;
                    _ = c.SDL_GetMouseState(&mx_i, &my_i);
                    const mx: f32 = @floatFromInt(mx_i); const my: f32 = @floatFromInt(my_i);
                    if (events.findScrollContainer(&root, mx, my)) |scroll_node| {
                        scroll_node.scroll_y -= @as(f32, @floatFromInt(event.wheel.y)) * 30.0;
                        const max_scroll = @max(0.0, scroll_node.content_height - scroll_node.computed.h);
                        scroll_node.scroll_y = @max(0.0, @min(scroll_node.scroll_y, max_scroll));
                    }
                },
                else => {},
            }
        }
